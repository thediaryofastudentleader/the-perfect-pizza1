import React, { createContext, useContext, useState, useCallback } from 'react'; import { PIZZA_MENU, EXTRA_TOPPINGS, TOPPING_PRICES } from '../data/menu'; interface PizzaState { appliedIngredients: string[]; matchedPizza: string | null; isCustom: boolean; basePrice: number; extrasPrice: number; totalPrice: number; showSuccess: boolean; orderDispatched: boolean; customerDetails: { name: string; phone: string; address: string }; } interface PizzaContextType extends PizzaState { addIngredient: (ingredient: string) => void; removeIngredient: (ingredient: string) => void; resetPizza: () => void; setCustomerDetails: (details: { name: string; phone: string; address: string }) => void; placeOrder: () => void; getHint: () => string; } const PizzaContext = createContext(null); export const usePizzaContext = () => { const context = useContext(PizzaContext); if (!context) throw new Error('usePizzaContext must be used within PizzaProvider'); return context; }; export const PizzaProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => { const [appliedIngredients, setAppliedIngredients] = useState([]); const [matchedPizza, setMatchedPizza] = useState(null); const [isCustom, setIsCustom] = useState(false); const [showSuccess, setShowSuccess] = useState(false); const [orderDispatched, setOrderDispatched] = useState(false); const [customerDetails, setCustomerDetails] = useState({ name: '', phone: '', address: '' }); const normalize = (s: string) => s.toLowerCase().trim(); const findMatch = useCallback((ingredients: string[]) => { const normalizedIngredients = ingredients.map(normalize); for (const item of PIZZA_MENU) { const normalizedMenuIngredients = item.ingredients.map(normalize); const allMenuInSelected = normalizedMenuIngredients.every(ing => normalizedIngredients.some(ni => ni.includes(ing) || ing.includes(ni)) ); const allSelectedInMenu = normalizedIngredients.every(ni => normalizedMenuIngredients.some(ing => ni.includes(ing) || ing.includes(ni)) ); if (allMenuInSelected && allSelectedInMenu) { return item; } } return null; }, []); const getHint = useCallback(() => { if (appliedIngredients.length === 0) return "Start by selecting ingredients from the menu or extras!"; const matchedItem = findMatch(appliedIngredients); if (matchedItem) { return `You've built a ${matchedItem.name}! Add extra toppings or checkout.`; } const partialMatches = PIZZA_MENU.filter(item => appliedIngredients.some(ing => item.ingredients.some(menuIng => normalize(menuIng).includes(normalize(ing)) || normalize(ing).includes(normalize(menuIng))) ) ); if (partialMatches.length > 0) { const closestMatch = partialMatches.reduce((prev, curr) => { const prevMatchCount = appliedIngredients.filter(ing => prev.ingredients.some(menuIng => normalize(menuIng) === normalize(ing)) ).length; const currMatchCount = appliedIngredients.filter(ing => curr.ingredients.some(menuIng => normalize(menuIng) === normalize(ing)) ).length; return currMatchCount > prevMatchCount ? curr : prev; }); const missingIngredients = closestMatch.ingredients.filter(ing => !appliedIngredients.some(ai => normalize(ai) === normalize(ing)) ); if (missingIngredients.length > 0) { return `Keep going! Try adding ${missingIngredients.join(', ')} to make a ${closestMatch.name}.`; } } return "Custom creation! Add more ingredients or checkout."; }, [appliedIngredients, findMatch]); const addIngredient = useCallback((ingredient: string) => { setAppliedIngredients(prev => { const newIngredients = [...prev, ingredient]; const matchedItem = findMatch(newIngredients); if (matchedItem) { setMatchedPizza(matchedItem.name); setIsCustom(false); setShowSuccess(true); setTimeout(() => setShowSuccess(false), 5000); } else { setMatchedPizza(null); setIsCustom(true); } return newIngredients; }); }, [findMatch]); const removeIngredient = useCallback((ingredient: string) => { setAppliedIngredients(prev => { const newIngredients = prev.filter(ing => ing !== ingredient); const matchedItem = findMatch(newIngredients); if (matchedItem) { setMatchedPizza(matchedItem.name); setIsCustom(false); } else { setMatchedPizza(null); setIsCustom(newIngredients.length > 0); } return newIngredients; }); }, [findMatch]); const resetPizza = useCallback(() => { setAppliedIngredients([]); setMatchedPizza(null); setIsCustom(false); setShowSuccess(false); }, []); const placeOrder = useCallback(() => { setOrderDispatched(true); resetPizza(); setTimeout(() => setOrderDispatched(false), 5000); }, [resetPizza]); const basePrice = matchedPizza ? (PIZZA_MENU.find(p => p.name === matchedPizza)?.price ?? 0) : 0; const menuItem = matchedPizza ? PIZZA_MENU.find(p => p.name === matchedPizza) : null; const extrasPrice = isCustom || (menuItem && appliedIngredients.length > menuItem.ingredients.length) ? appliedIngredients.reduce((sum, ing) => { if (menuItem && menuItem.ingredients.includes(ing)) return sum; return sum + (TOPPING_PRICES[ing] || 2.00); }, 0) : 0; const customBasePrice = isCustom && basePrice === 0 ? 12.00 : 0; const totalPrice = basePrice + extrasPrice + customBasePrice; return ( {children} ); };
